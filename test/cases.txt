# Filter

{{ foo | size: 3 }}

==>

Program(
  LiquidTemplate(
    Output(
      FilteredExpression(
        Expression(Identifier),
        Filter(
          Identifier,
          FilterArgs(
            FilterArg(
              Expression(Literal(Number))
            )
          )
        )
      )
    )
  )
)
