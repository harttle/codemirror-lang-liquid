# HTML

<div></div>
{}

==>

LiquidTemplate(HTML)

# Ouput

{{ foo }}

==>

LiquidTemplate(Output(Value(Identifier)))

# Multiple Ouputs

{{ foo }}{{ bar }}

==>

LiquidTemplate(
  Output(Value(Identifier))
  Output(Value(Identifier))
)

# Quoted

{{ "foo" }}
{{ "foo bar" }}
{{ "foo \" bar" }}

==>

LiquidTemplate(
  Output(Value(Literal(Quoted)))
  HTML
  Output(Value(Literal(Quoted)))
  HTML
  Output(Value(Literal(Quoted)))
)

# Range

{{ (1..3) }}

==>

LiquidTemplate(
  Output(
    Value(
      Range(
        Value(Literal(Number))
        Value(Literal(Number))
      )
    )
  )
)

# Range with Expressions

{{ (foo.bar..["foo"].size) }}

==>

LiquidTemplate(
  Output(
    Value(
      Range(
        Value(
          Identifier
          PropertyAccess(Identifier)
        )
        Value(
          SquareBracketProperty(Value(Literal(Quoted)))
          PropertyAccess(Identifier)
        )
      )
    )
  )
)

# Value

{{ "foo".size }}
{{ foo.bar["zoo"].coo }}
{{ ["zoo"].coo[0][bar] }}
{{ foo[bar.coo["arr[".size]][0] }}

==>

LiquidTemplate(
  Output(
    Value(
      Literal(Quoted),
      PropertyAccess(Identifier)
    )
  ),
  HTML,
  Output(
    Value(
      Identifier,
      PropertyAccess(Identifier),
      PropertyAccess(SquareBracketProperty(Value(Literal(Quoted)))),
      PropertyAccess(Identifier)
    )
  ),
  HTML,
  Output(
    Value(
      SquareBracketProperty(Value(Literal(Quoted))),
      PropertyAccess(Identifier),
      PropertyAccess(SquareBracketProperty(Value(Literal(Number)))),
      PropertyAccess(SquareBracketProperty(Value(Identifier)))
    )
  )
  HTML,
  Output(
    Value(
      Identifier,
      PropertyAccess(
        SquareBracketProperty(
          Value(
            Identifier
            PropertyAccess(Identifier),
            PropertyAccess(
              SquareBracketProperty(
                Value(
                  Literal(Quoted)
                  PropertyAccess(Identifier)
                )
              )
            )
          )
        )
      )
      PropertyAccess(
        SquareBracketProperty(
          Value(Literal(Number))
        )
      )
    )
  )
)

# Filter

{{ foo | size: 3 }}

==>

LiquidTemplate(
  Output(
    Value(Identifier),
    Filter(
      Identifier,
      FilterArgs(
        FilterArg(
          Value(Literal(Number))
        )
      )
    )
  )
)

# Multiple Filters

{{ person | title: "Sr." | prepend: "Sir" }}

==>

LiquidTemplate(
  Output(
    Value(Identifier),
    Filter(
      Identifier,
      FilterArgs(FilterArg(Value(Literal(Quoted))))
    ),
    Filter(
      Identifier,
      FilterArgs(FilterArg(Value(Literal(Quoted))))
    )
  )
)

# Multiple Argument Filter

{{ person | card: "Sr.", age:18, report: mgrs[0].name }}

==>

LiquidTemplate(
  Output(
    Value(Identifier),
    Filter(
      Identifier,
      FilterArgs(
        FilterArg(Value(Literal(Quoted))),
        FilterArg(KeyValuePair(Identifier,Value(Literal(Number)))),
        FilterArg(
          KeyValuePair(
            Identifier,
            Value(
              Identifier,
              PropertyAccess(SquareBracketProperty(Value(Literal(Number)))),
              PropertyAccess(Identifier)
            )
          )
        )
      )
    )
  )
)

)

# Tag

{% foo %}
{% endfoo %}

==>

LiquidTemplate(

    TagName(Identifier)
  )
  HTML

    TagName(Identifier)
  )
)

# Tag with Arguments

{% foo reverse col : 3 row:rows[0] %}

==>

LiquidTemplate(
  Tag(
    TagName(Identifier)
    TagArg(Identifier),
    TagArg(KeyValuePair(Identifier,Value(Literal(Number)))),
    TagArg(KeyValuePair(
      Identifier,
      Value(
        Identifier,
        PropertyAccess(SquareBracketProperty(Value(Literal(Number))))
      )
    ))
  )
)
