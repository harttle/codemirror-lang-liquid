# HTML

<div></div>
{}

==>

LiquidTemplate(HTML)

# Ouput

{{ foo }}

==>

LiquidTemplate(Output(FilteredExpression(Value(Identifier))))

# Filter

{{ foo | size: 3 }}

==>

LiquidTemplate(
  Output(
    FilteredExpression(
      Value(Identifier),
      Filter(
        Identifier,
        FilterArgs(
          FilterArg(
            Value(Literal(Number))
          )
        )
      )
    )
  )
)

# Boolean

{{ true }}
{{ false }}

==>

LiquidTemplate(
  Output(FilteredExpression(Value(Literal(Boolean))))
  HTML
  Output(FilteredExpression(Value(Literal(Boolean))))
)

# Property Access

{{ foo.bar }}

==>

LiquidTemplate(
  Output(
    FilteredExpression(
      Value(
        Identifier,
        PropertyAccess(Identifier)
      )
    )
  )
)

# Square Bracket Property Access

{{ foo["bar"] }}

==>

LiquidTemplate(
  Output(
    FilteredExpression(
      Value(
        Identifier,
        PropertyAccess(
          SquareBracketProperty(
            Value(Literal(Quoted))
          )
        )
      )
    )
  )
)
